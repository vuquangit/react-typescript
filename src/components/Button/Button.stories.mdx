# Button

With `MDX` we can define a story for `Button` right in the middle of our
Markdown documentation.

export const Template = (args) => <Button {...args} />

<Canvas>
  <Story
    name="Normal"
    args={{ variant: 'normal' }}
    onClick={action('Medium button clicked')}
  >
    {Template.bind({})}
  </Story>
  <Story
    name="Large"
    args={{ variant: 'large' }}
    onClick={action('Large button clicked')}
  >
    {Template.bind({})}
  </Story>
  <Story
    name="Disable"
    args={{ disabled: true }}
    onClick={action('Disable button clicked')}
  >
    {Template.bind({})}
  </Story>
</Canvas>
